    ```sql
    -- Create new database
    CREATE DATABASE IF NOT EXISTS `xeno_shopify` DEFAULT CHARACTER SET utf8mb4;
    -- Check using this command:
    SHOW DATABASES;
    -- Use the database
    USE xeno_shopify;
    -- Check tables
    SHOW TABLES;
    ```
